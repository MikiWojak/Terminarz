/*Skrócone zapytanie złożone*/
SELECT g.nazwa_grupa, z.tytul_zadanie, z.data_zadanie, z.czy_wykonane 
FROM grupy AS g, zadania AS z, przypisania AS p 
WHERE g.id_grupa = p.id_grupa AND z.id_zadanie = p.id_zadanie



/*Szczegółowe zapytanie złożone*/
SELECT z.id_zadanie, z.data_zadanie, z.tytul_zadanie, z.opis_zadanie, z.priorytet_zadanie, z.czy_wykonane, g.id_grupa, g.nazwa_grupa, g.opis_grupa
FROM zadania AS z, grupy AS g, przypisania AS p
WHERE z.id_zadanie = p.id_zadanie AND g.id_grupa = p.id_grupa


/*Zapytanie złożone o zadania należące do danej grupy*/
SELECT z.id_zadanie, z.data_zadanie, z.tytul_zadanie, z.opis_zadanie, z.priorytet_zadanie, z.czy_wykonane, g.id_grupa, g.nazwa_grupa, g.opis_grupa
FROM zadania AS z, grupy AS g, przypisania AS p
WHERE z.id_zadanie = p.id_zadanie AND g.id_grupa = p.id_grupa AND p.id_grupa = 1



/*Zapytanie złożone o grupy przypisane do zadania*/
SELECT g.id_grupa, g.nazwa_grupa
FROM zadania AS z, grupy AS g, przypisania AS p
WHERE z.id_zadanie = p.id_zadanie AND g.id_grupa = p.id_grupa AND p.id_zadanie = 2
